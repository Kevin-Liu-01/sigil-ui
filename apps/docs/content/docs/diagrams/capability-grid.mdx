## Import

```tsx
import { CapabilityGrid } from "@sigil-ui/components";
```

## Usage

```tsx
<CapabilityGrid
  categories={[
    { label: "Core", items: [{ name: "Tokens" }, { name: "Presets" }] },
    { label: "Surface", items: [{ name: "Components" }, { name: "Docs" }] },
  ]}
/>
```

## Props

| Prop | Type | Default | Description |
|---|---|---|---|
| `categories` | `CapabilityCategory[]` | — | Categories with items |
| `callout` | `CapabilityCallout` | — | Optional callout section |
