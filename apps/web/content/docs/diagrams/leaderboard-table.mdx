## Usage

```tsx
<LeaderboardTable
  title="Top Contributors"
  valueLabel="Points"
  rows={[
    { label: "Alice", value: 2480, delta: "+120" },
    { label: "Bob", value: 2310, delta: "+85" },
    { label: "Carol", value: 2150, delta: "-30" },
    { label: "Dave", value: 1980, delta: "+200", highlight: true },
    { label: "Eve", value: 1870, delta: "+10" },
  ]}
/>
```
